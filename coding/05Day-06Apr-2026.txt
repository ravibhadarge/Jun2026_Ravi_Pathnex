🔹 Ansible Task — Install Nginx on Pathnex Server
Rewrite this YAML manually:
---
---
- name: Install and Start Nginx
  hosts: all
  become: true
  tasks:
    - name: Install Nginx (Ubuntu)
      apt:
        name: nginx
        state: present
        update_cache: yes
      when: ansible_os_family == "Debian"

    - name: Install Nginx (RHEL/Fedora)
      yum:
        name: nginx
        state: present
      when: ansible_os_family == "RedHat"

    - name:  Nginx service is started and enabled
      service:
        name: nginx
        state: started
        enabled: true

- name: Install ZSH on Ubuntu and CentOS
  hosts: all
  become: true

  tasks:

    # Ubuntu / Debian
    - name: Update apt cache
      apt:
        update_cache: yes
        cache_valid_time: 3600
      when: ansible_os_family == "Debian"

    - name: Install ZSH on Ubuntu
      apt:
        name: zsh
        state: present
      when: ansible_os_family == "Debian"

    # CentOS / RedHat
    - name: Install EPEL repo
      yum:
        name: epel-release
        state: present
      when: ansible_os_family == "RedHat"

    - name: Install ZSH on CentOS
      yum:
        name: zsh
        state: present
      when: ansible_os_family == "RedHat"
